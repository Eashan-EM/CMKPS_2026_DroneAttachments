Installation of web app in pi 
code:(install dependencies) 
pip3 install flask 
code: (creating of folder)
mkdir ~/droneapp && cd ~/droneapp
mkdir templates 
code:  (copying yolo files into the folder)
cp ~/yolov3-tiny.weights ~/droneapp/
cp ~/yolov3-tiny.cfg ~/droneapp/
cp ~/coco.names ~/droneapp/
code: (creating of app.py)
cat > ~/droneapp/app.py << 'EOF'
from flask import Flask, render_template, jsonify
import cv2
import numpy as np
import threading
import time

app = Flask(__name__)

person_count = 0
lock = threading.Lock()

net = cv2.dnn.readNet("yolov3-tiny.weights", "yolov3-tiny.cfg")
with open("coco.names", "r") as f:
    classes = [line.strip() for line in f.readlines()]
layer_names = net.getUnconnectedOutLayersNames()

def detect_persons():
    global person_count
    cap = cv2.VideoCapture(0, cv2.CAP_V4L2)
    cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
    cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)

    while True:
        ret, frame = cap.read()
        if not ret:
            time.sleep(0.1)
            continue

        blob = cv2.dnn.blobFromImage(frame, 1/255.0, (416, 416),
                                      swapRB=True, crop=False)
        net.setInput(blob)
        outputs = net.forward(layer_names)

        count = 0
        for output in outputs:
            for detection in output:
                scores = detection[5:]
                class_id = np.argmax(scores)
                confidence = scores[class_id]
                if confidence > 0.5 and classes[class_id] == "person":
                    count += 1

        with lock:
            person_count = count

        time.sleep(0.1)

@app.route('/')
def index():
    return render_template('index.html')

@app.route('/count')
def count():
    with lock:
        return jsonify({"persons": person_count})

if __name__ == '__main__':
    t = threading.Thread(target=detect_persons, daemon=True)
    t.start()
    app.run(host='0.0.0.0', port=5000, debug=False)
EOF

<!DOCTYPE html>
<html>
<head>
  <title>Drone Detection</title>
  <style>
    body {

      font-weight: bold;
      color: #00ff88;
    }
    #status { color: #555; margin-top: 20px; font-size: 0.9em; }
  </style>
</head>
<body>
  <h1>Persons Detected</h1>
  <div id="count">--</div>
  <div id="status">connecting...</div>

  <script>
    function fetchCount() {
      fetch('/count')
        .then(res => res.json())
        .then(data => {
          document.getElementById('count').innerText = data.persons;
          document.getElementById('status').innerText =
            'Last updated: ' + new Date().toLocaleTimeString();
        })
        .catch(() => {
          document.getElementById('status').innerText = 'disconnected';
        });
    }
    fetchCount();
    setInterval(fetchCount, 1000);
  </script>
</body>
</html>
EOF
code: (setting up pi 4 as hotspot)
sudo apt install hostapd dnsmasq -y
sudo systemctl stop hostapd dnsmasq
sudo nano /etc/hostapd/hostapd.conf
add the below mentioned code at the bottom: 
(interface=wlan0
driver=nl80211
ssid=DronePI4
hw_mode=g
channel=7
wpa=2
wpa_passphrase=drone1234
wpa_key_mgmt=WPA-PSK
rsn_pairwise=CCMP
)
sudo nano /etc/dnsmasq.conf
add the below mentioned code at the bottom:
(interface=wlan0
dhcp-range=192.168.4.2,192.168.4.20,255.255.255.0,24h)
sudo nano /etc/dhcpcd.conf
add the code mentioned below at the bottom:
(interface=wlan0
static ip_address=192.168.4.1/24
nohook wpa_supplicant)
code: (enabling the app)
sudo systemctl unmask hostapd
sudo systemctl enable hostapd dnsmasq
sudo reboot
code: (run the app)
cd ~/droneapp
python3 app.py
